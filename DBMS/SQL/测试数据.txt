SELECT * FROM STUDENT;

SELECT * FROM STUDENT,CHOOSE;
SELECT * FROM STUDENT,CHOOSE WHERE (SSEX=0) AND (CID=1);
SELECT * FROM STUDENT,CHOOSE WHERE (SSEX=1) AND (CID=1);
